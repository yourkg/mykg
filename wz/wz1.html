#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int s = 0, k, i, j, t, x, f[9], g[9];
	long a, y;
	for (a = 12345678; a <= 87654321; a += 9)
	{
		y = a; k = 0; //cout << "y=a=" << a<<endl; +9的原因是对于10进制能够快速进位（对棋子换行）
		for (i = 1; i <= 8; i++)f[i] = 0;
		while (y > 0)          //此算法的妙处就是直接将题意《必不处于同行同列》这一条件化成处理没有重复数字的8位数
		{
			x = y % 10;
			f[x] += 1;     //（与下面！第一步筛选！处结合 筛选符合条件的y）符合条件既能反应8*8方格的mom位置。
			y = y / 10;    //并将其倒叙放入g【】 对于这些情况 不用在乎g与原来a所表示的对应
			k++;
			g[k] = x;//g[1]=8  7 6 5 4 3 2 1
		}//for (i = 1; i <= 8; i++)cout << f[i] << " ";
		//cout << endl;
		//for (i = 1; i <= 8; i++)cout << g[i] << " ";
		//cout << endl<<endl<<"------------------------------\n";
	//}

	   for (t = 0, i = 1; i <= 8; i++)    //！第一步筛选！
			if (f[i] != 1)t = 1;         //   与上面的到的f【】处理，
		if (t == 1)continue;             //判断是否进行下面对g【】的继续筛选
		for (k = 1; k <= 7; k++)           //------------------------
			for (j = k + 1; j <= 8; j++)   //          第二部              -
				if (fabs(g[j] - g[k]) == j - k)t = 1;//     筛选掉在斜线上的 根据在斜线上的特点行列差相等            -
		if (t == 1)continue;              //------------------------------
		s++;        //执行到这一步记录一个符合条件的情况。。
		cout << a<<endl;
		if (s % 6 == 0)cout << endl;
	}
	cout << "\n s=" << s;
	system("pause");
	return 0;
}
